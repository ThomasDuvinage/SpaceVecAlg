82  constexpr T taylor_2_bound = details::sqrt(taylor_0_bound);
83  constexpr T taylor_n_bound = details::sqrt(taylor_2_bound);
84 
85  // We use the 4th order taylor series around 0 of x/sin(x) to compute
86  // this function:
87  //
88  // x^2 7x^4
89  // 1 + ── + ──── + O(x^6)
90  // 6 360
91  // this approximation is valid around 0.
92  // if x is far from 0, our approximation is not valid
93  // since x^6 becomes non neglectable we use the normal computation of the function
94  // (i.e. taylor_2_bound^6 + taylor_0_bound == taylor_0_bound but
95  // taylor_n_bound^6 + taylor_0_bound != taylor_0).
96 
97  if(std::abs(x) >= taylor_n_bound)
98  {
99  return (x / std::sin(x));
100  }
101  else
102  {
103  // x is bellow taylor_n_bound so we don't care of the 6th order term of
104  // the taylor series.

106  T result = static_cast<T>(1);
107 
108  if(std::abs(x) >= taylor_0_bound)
109  {
